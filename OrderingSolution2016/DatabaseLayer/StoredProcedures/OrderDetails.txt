ALTER PROCEDURE [dbo].[OrderDetails]
	
	(
	@OrderID int 
	)
	
AS
 	
 SET NOCOUNT ON;

 SELECT ProductName, [Order Details].*
 FROM [Order Details] Inner Join Products on [Order Details].ProductID = Products.ProductID
 WHERE OrderID = @OrderID;

	RETURN
